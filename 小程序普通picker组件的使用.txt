微信小程序上打算做一个性别下拉框。

wxml代码：
  <view class="section">
    <text>性别：</text>
    <picker mode="selector"
            range="{{xingbie}}"
            value="{{index}}"
            bindchange="changeSex"
    >
       {{xingbie[index]}}
    </picker>
  </view>
  
js代码：
Page({
  data: {
    xingbie:[
       "男","女"
    ],
    index:0,          //0为男生，1为女生
  },
  /**
   * 监听性别picker选择器的改变
   */
  changeSex: function(e) {
    this.setData({
       index:e.detail.value,
    });
  }

})
